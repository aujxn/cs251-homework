\begin{homeworkProblem}
   Convert the following into first order predicate calculus:

   \begin{itemize}
   \item For any two real numbers, there is a real number between them.\\
   $\forall x, y \in \mathbb{R}.\exists z \in \mathbb{R}.x<z<y \lor x>z>y$

   \item A natural number is prime if, and only if, it has no divisors other than 1 and itself.\\
      $\forall p \in \mathbb{N}. p \in P \iff (\nexists d \in \{x \in \mathbb{N} \mid x \neq p \land x \neq 1\}.d \mid p)$

   \item A Language is Turing Complete if it can encode Turing Machines $L = TM$.
      There is a compiler between any two Turing Complete languages.\\
      L is the set of all languages.\\
         $\forall x,y \in L. TuringComp(x) \land TuringComp(y) \to (\exists f: x \to y)$

   \item The set of closed balls in $n$ dimensional space is $B^n$.
      A fixed point is a point that doesn't change after applying a function.
      Any continuous function on a closed ball has at least one fixed point.\\
         $\forall n \in \mathbb{N}. \forall f: B^n \to B^n. Continuous(f) \to (\exists b \in B^n. b \equiv f(b))$

   \item Every planer graph is 4-colorable.\\
      $\forall g \in Graphs. Planer(g) \to Colorable(g, 4)$
   \end{itemize}

\end{homeworkProblem}

\begin{homeworkProblem}
   Restate the following in english:

   \begin{itemize}
   \item $\forall n,a,b \in \mathbb{N}.n\mid a \land n \mid b \to \exists x,y \in \mathbb{N}.n \mid ax+by$\\
      For every three natural numbers, if one divides the other two, there are two natural scalars that scale the
      dividends  such that the divisor divides the sum of the scaled values.

   \item $\forall n \in \mathbb{N}. n > 2 \to (\nexists a, b, c \in \mathbb{N}. a^n + b^n = c^n)$\\
      Fermat's last theorem. There are no natural solutions that satisfy $a^n + b^n = c^n$ larger than $n = 2$ for
      every natural $a,b,c$.

   \item $\nexists p, q \in \mathbb{N}. \frac{p}{q}^2 \neq 2$\\
      The square root of two is not a rational number.

   \item $\exists x \in \mathbb{R}. x^2 + 1 = 0$\\
      There is a real number that is also imaginary.

   \end{itemize}
\end{homeworkProblem}

\begin{homeworkProblem}
   Determine if the following are well formed (legal). P is a predicate with arity 3, Q is a predicate with arity 2,
   and $f$ is a function with arity 2, and g is a function with arity 1. The only domain is $\mathbb{N}$.

   \begin{itemize}
      \item $\forall x.P(f(x, x), g(x), x) = x$\\
         No, predicate P is a true or false value and cannot be compared to x, a natural.

      \item $\forall x.(\exists y.Q(x,y) \land P(x,y,z)) \to f(x,y) = g(x)$\\
         Well formed. Both sides of the implication evaluate to a boolean value.

      \item $\nexists y.y = f(y,y)$\\
         Well formed.

      \item $\forall x,y,z.P(x, y, z) \to Q(x,y,z)$\\
         No, Q has an arity of 2.

      \item $\forall z.1+1=2$\\
         Pretty useless but well formed.
   \end{itemize}
\end{homeworkProblem}

\begin{homeworkProblem}
   Reduce the following to shortest form.

   \begin{itemize}
      \item $\nexists x \exists y.(\lnot x \lor \lnot y)$
         \begin{align*}
            &\forall x. \forall y \lnot(\lnot x \lor \lnot y)    &&DM_{\exists} * 2\\
            &\forall x, y. x \land y                             &&DM_{\lor}
         \end{align*}
      \item $\forall q.((\forall x.x) \land (\forall y.y) \to (\forall z.q))$
         \begin{align*}
            &\forall q,z.((\forall x.x) \land (\forall y.y) \to q)    &&Cov_{\forall}\\
            &\forall q,z.(\forall x,y.x \land y) \to q    &&Com_{\land}, \forall_{\land} * 2\\
            &\forall q,z \exists x,y.x \land y \to q        &&Cont_{\forall}
         \end{align*}
      \item $(\forall x.x) \to (\exists y. \lnot y)$
         \begin{align*}
            &\exists x.x \to (\exists y. \lnot y) &&Cont_{\forall}\\
            &\exists x,y.x \to \lnot y    &&Cov_{\exists}
         \end{align*}
   \end{itemize}
\end{homeworkProblem}

\begin{homeworkProblem}
   Convert the following into PNF.

   \begin{itemize}
      \item $(\forall a \exists b. a \to b) \land (\forall b \exists a. a \to b)$
         \begin{align*}
            &(\forall a \exists b. a \to b) \land (\forall c \exists d. a \to b[b \mapsto c][a \mapsto d])   &&Rep_{\forall}, Rep_{\exists}\\
            &\forall a,c \exists b,d. a \to b \land d \to c      &&\forall_{\land}, \exists_{\land}
         \end{align*}
      \item $\forall a.P(a) \to (\exists a.R(a,a))$
         \begin{align*}
            &\forall a.P(a) \to (\exists b.R(a,a)[a \mapsto b])      &&Rep_{\exists}\\
            &\forall a \exists b.P(a) \to R(b,b)         &&Cov_{\exists}
         \end{align*}
      \item $\forall x.(\forall y.x \to y) \land (\forall y.y \to x)$
         \begin{align*}
            &\forall x.(\forall y.x \to y) \land (\forall z.y \to x[y \mapsto z])      &&Rep_{\forall}\\
            &\forall x,y,z.x \to y \land z \to x               &&\forall_{\land} * 2
         \end{align*}
      \item $\forall x.(\forall y.x \to y) \to (\forall z.z)$
         \begin{align*}
            &\forall x, z \exists y.(x \to y) \to z   &&Cont_{\forall}, Cov_{\exists}\\
         \end{align*}
      \item $\forall x. \lnot (\forall y.y) \to \lnot (\forall z. x \land z)$
         \begin{align*}
            &\forall x.(\exists y. \lnot y) \to (\exists z.\lnot (x \land z))    &&DM_{\exists}\\
            &\forall x,y \exists z.\lnot y \to \lnot (x \land z)     &&Cont_{\exists}, Cov_{\exists}\\
         \end{align*}
   \end{itemize}
\end{homeworkProblem}
